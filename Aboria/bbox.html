Struct template bbox

Aboria::bbox — Contains the minimum and maximum extents of a hypercube in D dimensional space.

Synopsis

// In header: </home/travis/build/martinjrobins/Aboria/src/SpatialUtil.h>

template<unsigned int D> 
struct bbox {
  // types
  typedef Vector< double, D > double_d;

  // construct/copy/destruct
  bbox();
  bbox(const double_d &);
  bbox(const double_d &, const double_d &);

  // public member functions
  bbox operator+(const bbox &);
  bool operator<(const bbox &);
  bool operator<=(const bbox &);
  bool is_empty();

  // public data members
  double_d bmin;  // minimum point in the box (i.e. lower left corner for D=2) 
  double_d bmax;  // maximum point in the box (i.e. upper right corner for D=2) 
};
